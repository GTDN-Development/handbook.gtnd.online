## API Reference

The component extends `<motion.div>` and adding following props on top.

<TypeTable
  type={{
    start: {
      description: "Starting number for the counter",
      type: "number",
      required: true,
    },
    end: {
      description: "Ending number for the counter",
      type: "number",
      required: true,
    },
    decimals: {
      description: "Number of decimal places to display",
      type: "number",
      default: "2",
    },
    duration: {
      description: "Duration of the animation in seconds",
      type: "number",
      default: "2",
    },
    delay: {
      description: "Delay before animation starts in seconds",
      type: "number",
      default: "0",
    },
    ease: {
      description: "Easing function for the animation",
      type: 'KeyframeOptions["ease"]',
      default: "[0.25, 1, 0.5, 1]",
    },
  }}
/>
