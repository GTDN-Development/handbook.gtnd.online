## Usage

```tsx
import { chain } from "@/lib/utils";

function Button({ onClick }: { onClick?: () => void }) {
  const handleClick = () => console.log("Internal handler");
  const trackAnalytics = () => console.log("Track click");
  
  return (
    <button onClick={chain(handleClick, trackAnalytics, onClick)}>
      Click me
    </button>
  );
}
// All three handlers execute on click (onClick safely ignored if undefined)
```

<Steps>
<Step>

```bash
touch src/lib/utils.ts
```
</Step>

<Step> 
**Copy & paste the source code to your lib/utils.ts file**
        ```tsx
        export function chain(...callbacks: any[]): (...args: any[]) => void {
          return (...args: any[]) => {
            for (let callback of callbacks) {
              if (typeof callback === "function") {
                callback(...args);
              }
            }
          };
        }
        ```
</Step>
