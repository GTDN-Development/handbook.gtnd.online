## Usage

```tsx
import { formatDate } from "@/lib/utils";
```

```tsx
// Basic usage with default Czech locale
formatDate(new Date()); // "25. pro 2024"

// With custom language
formatDate(new Date(), "en-US"); // "Dec 25, 2024"

// With custom options
formatDate(new Date(), "cs-CZ", { 
  year: "numeric", 
  month: "long", 
  day: "numeric" 
}); // "25. prosince 2024"

// Works with date strings too
formatDate("2024-12-25"); // "25. pro 2024"
```

## Installation

<Step>
**Copy & paste the source code to your lib/utils.ts file**

```tsx title="lib/utils.ts"
export function formatDate(
  date: Date | string,
  language: string = "cs-CZ",
  options?: Intl.DateTimeFormatOptions
): string {
  return new Date(date).toLocaleDateString(
    language,
    options || { year: "numeric", month: "short", day: "numeric" }
  );
}
```
</Step>
</Steps>
