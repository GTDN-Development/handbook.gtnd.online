## Overview

Before installing Header and Footer components, you need to set up the shared configuration files and dependencies that these layout components require.

## Installation

<Steps>
    <Step>
        **Create the NavLink component**

        The NavLink component is used for navigation links with active page detection and external link handling.

        ```bash
        touch src/components/layout/nav-link.tsx
        ```

        ```tsx title="src/components/layout/nav-link.tsx"
        "use client";

        import { Link, type LinkProps } from "../ui/link";
        import { usePathname } from "next/navigation";
        import { ArrowUpRightIcon } from "lucide-react";

        export function NavLink({
          children,
          href,
          target,
          rel,
          showExternalIcon = false,
          ...props
        }: LinkProps & { showExternalIcon?: boolean }) {
          const pathname = usePathname();
          const isExternal = typeof href === "string" && href.startsWith("http");
          const isCurrent = pathname === href;

          return (
            <Link
              {...props}
              href={href}
              aria-current={isCurrent ? "page" : undefined}
              data-current={isCurrent ? "true" : undefined}
              data-external={isExternal ? "true" : undefined}
              target={isExternal ? target || "_blank" : target}
              rel={isExternal ? rel || "noopener noreferrer" : rel}
            >
              {children}
              {isExternal && showExternalIcon && (
                <ArrowUpRightIcon aria-hidden="true" className="ml-1 inline size-[1em] opacity-50" />
              )}
            </Link>
          );
        }
        ```
    </Step>

    <Step>
        **Create navigation config files**

        These configuration files define your site's navigation structure, legal links, and social media links.

        ```bash
        touch src/config/legal-links.ts
        touch src/config/nav-links.ts
        touch src/config/social-media-links.ts
        ```

        ```ts title="src/config/legal-links.ts"
        import type { NavigationLink } from "./nav-links";

        export const legalLinks = {
          gdpr: { name: "Privacy Policy", href: "/gdpr" },
          cookies: { name: "Cookie Policy", href: "/cookies" },
        } as const satisfies Record<string, NavigationLink>;

        export const legalLinksArray: NavigationLink[] = Object.values(legalLinks);
        ```

        ```ts title="src/config/nav-links.ts"
        import { legalLinksArray } from "./legal-links";
        import type { Route } from "next";

        export type NavigationLink<T extends string = string> = {
          name: string;
          href: Route<T>;
        };

        export type NavigationDropdown = {
          name: string;
          items: NavigationLink[];
        };

        export type NavigationItem = NavigationLink | NavigationDropdown;

        export const navLinks = {
          home: {
            name: "Home",
            href: "/",
          },
          legalLinks: {
            name: "Legal links",
            items: [...legalLinksArray],
          },
          contact: {
            name: "Contact",
            href: "/contact",
          },
        } as const satisfies Record<string, NavigationItem>;

        export const navLinksArray: NavigationItem[] = Object.values(navLinks);
        ```

        ```ts title="src/config/social-media-links.ts"
        import { InstagramIcon, TwitterXIcon, YoutubeIcon, GitHubIcon } from "@/components/ui/icon-brand";
        ```

    <Step>
        **Create the SocialMediaIcons component**

        This component renders social media links as icon buttons.

        ```bash
        touch src/components/layout/social-media-icons.tsx
        ```

        ```tsx title="src/components/layout/social-media-icons.tsx"
        import { socialMediaLinksArray } from "@/config/social-media-links";
        import { cn } from "@/lib/utils";
        import { Button } from "../ui/button";

        export function SocialMediaIcons(props: React.ComponentProps<"ul">) {
          return (
            <ul
              {...props}
              className={cn(
                "flex flex-wrap items-center justify-between gap-2",
                props.className,
              )}
            >
              {socialMediaLinksArray.map((item, index) => (
                <li key={index}>
                  <Button size="icon" variant="ghost" asChild>
                    <a href={item.href} target="_blank" rel="noopener noreferrer">
                      <span className="sr-only">{item.name}</span>
                      <item.icon aria-hidden="true" />
                    </a>
                  </Button>
                </li>
              ))}
            </ul>
          );
        }
        ```
